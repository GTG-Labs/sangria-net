## 2. Initialize the client

```ts
// lib/sangria.ts
import { Sangria } from "@sangria-sdk/core";

export const sangria = new Sangria({
  apiKey: process.env.SANGRIA_SECRET_KEY!,
});
```

## 3. Set a price on your endpoint

Create your API route handler like you normally would. Then wrap it with `fixedPrice` — this charges the caller before your handler runs.

```ts
// app/api/premium/route.ts
import { NextRequest, NextResponse } from "next/server";
import { sangria } from "@/lib/sangria";
import { fixedPrice } from "@sangria-sdk/core/nextjs";

async function handler(request: NextRequest) {
  return NextResponse.json({ message: "Premium content" });
}

export const GET = fixedPrice(
  sangria,
  { price: 0.01, description: "Access premium content" },
  handler
);
```

### Bypassing payment

If your customers already authenticate with their own API key, you can skip Sangria payment for those requests:

```ts
export const GET = fixedPrice(
  sangria,
  { price: 0.01, description: "Access premium content" },
  handler,
  {
    bypassPaymentIf: (request) => {
      const key = request.headers.get("x-api-key");
      return key != null && isValidApiKey(key); // validate against your own auth system
    },
  }
);
```

### Accessing payment data

Use the `getSangria` helper to read payment details inside your handler:

```ts
import { fixedPrice, getSangria } from "@sangria-sdk/core/nextjs";

async function handler(request: NextRequest) {
  const payment = getSangria(request);
  // payment.paid — true if agent paid
  // payment.amount — the price charged
  // payment.transaction — on-chain tx hash

  return NextResponse.json({ paid: payment?.paid });
}
```

## Handle errors

If the Sangria backend is *unreachable, times out, or returns a non-2xx status*, the SDK throws a `SangriaError`. Next.js App Router doesn't have global error middleware for API routes, so wrap the call in a try/catch:

```ts
// app/api/premium/route.ts
import { NextRequest, NextResponse } from "next/server";
import { sangria } from "@/lib/sangria";
import { SangriaError } from "@sangria-sdk/core";
import { fixedPrice } from "@sangria-sdk/core/nextjs";

async function handler(request: NextRequest) {
  return NextResponse.json({ message: "Premium content" });
}

const premiumRoute = fixedPrice(
  sangria,
  { price: 0.01, description: "Access premium content" },
  handler
);

export async function GET(request: NextRequest) {
  try {
    return await premiumRoute(request);
  } catch (err) {
    if (err instanceof SangriaError) {
      console.error(`[sangria:${err.operation}]`, err.message);
      return NextResponse.json(
        { error: "Payment provider unavailable" },
        { status: 503 }
      );
    }
    throw err;
  }
}
```

`SangriaError` has an `.operation` field (`"generate"` or `"settle"`) so you know which step failed. Subclasses give you more detail:

| Error | When | Properties |
|-------|------|------------|
| `SangriaConnectionError` | Network failure (DNS, refused, reset) | — |
| `SangriaTimeoutError` | Request timed out | — |
| `SangriaAPIStatusError` | Backend returned 4xx/5xx | `.statusCode`, `.response` |

Business-level payment failures (bad signature, insufficient funds) are **not** errors — they return a normal `402` response to the caller.

## That's it

Your endpoint is now priced. When an AI agent calls it, here's what happens:

1. The SDK returns a `402` with the price and payment details
2. The agent signs a USDC payment off-chain and retries with a `payment-signature` header
3. The SDK settles the payment on-chain, and your handler runs
